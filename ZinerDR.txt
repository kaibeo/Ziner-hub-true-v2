local function findSwingRemote()
    local path = { "Shared", "Universe", "Network", "RemoteEvent", "SwingMelee" }
    local obj = ReplicatedStorage
    for _, part in ipairs(path) do
        obj = obj:FindFirstChild(part)
        if not obj then
            return nil
        end
    end
    return obj
end

local swingRemote = findSwingRemote()
local allowedWeapons = {
    "Shovel",
    "Sword",
    "Pickaxe",
    "Vampire Knife",
    "Jade Sword",
    "Tomahawk",
    "Excalibur",
    "Cavalry Sword",
    "Spear",
    "Spear of Longinus"
}

local function startFiring(tool)
    if firing then return end
    firing = true
    currentTask = task.spawn(function()
        while firing do
            swingRemote = swingRemote or findSwingRemote()
            if swingRemote then
                local args = {
                    tool,
                    1767491059.30076,
                    Vector3.new(0.409719854593277, -0.543113112449646, 0.7329104542732239)
                }
                swingRemote:FireServer(unpack(args))
            end
            task.wait()
        end
    end)
end

local function stopFiring()
    firing = false
    if currentTask then
        task.cancel(currentTask)
        currentTask = nil
    end
end

local function hookTool(tool)
    if tool:IsA("Tool") then
        local equipConn = tool.Equipped:Connect(function()
            startFiring(tool)
        end)
        local unequipConn = tool.Unequipped:Connect(function()
            stopFiring()
        end)
        table.insert(autohitConnections, equipConn)
        table.insert(autohitConnections, unequipConn)
    end
end

local function hookAllTools()
    local character = player.Character or player.CharacterAdded:Wait()
    for _, weaponName in ipairs(allowedWeapons) do
        local tool = character:FindFirstChild(weaponName) or player.Backpack:FindFirstChild(weaponName)
        if tool then hookTool(tool) end
    end
    table.insert(autohitConnections, character.ChildAdded:Connect(function(child)
        if table.find(allowedWeapons, child.Name) then
            hookTool(child)
        end
    end))
    table.insert(autohitConnections, player.Backpack.ChildAdded:Connect(function(child)
        if table.find(allowedWeapons, child.Name) then
            hookTool(child)
        end
    end))
end

local function disconnectAll()
    for _, conn in ipairs(autohitConnections) do
        pcall(function() conn:Disconnect() end)
    end
    autohitConnections = {}
    stopFiring()
end

local function enableAutoHit()
    if not autohitActive then
        autohitActive = true
        hookAllTools()
        updateButtonVisual()
    end
end

local function disableAutoHit()
    if autohitActive then
        autohitActive = false
        disconnectAll()
        updateButtonVisual()
    end
end

local function toggleAutoHit()
    if autohitActive then
        disableAutoHit()
    else
        enableAutoHit()
    end
end

button.MouseButton1Click:Connect(function()
    toggleAutoHit()
end)

UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    if input.KeyCode == autohitKey then
        toggleAutoHit()
    end
end)

local Players = game:GetService("Players")

local CONFIG = {
    enabled = false,
    radius = 50,
    height = 2,
}

local MOB_FOLDER_NAMES = {
    "Zombies",
    "Animals", 
    "Enemies",
    "RuntimeEntities",
    "NightEnemies",
    "Mobs",
    "NPCs"
}

local currentTarget = nil
local bodyPosition = nil
local bodyGyro = nil
local connection = nil

local cachedFolders = {}
local lastFolderScan = 0
local lastTargetSearch = 0
local heightCache = {}

local FOLDER_SCAN_INTERVAL = 3
local TARGET_SEARCH_INTERVAL = 0.15

local function findMobFolders()
    local now = tick()
    if now - lastFolderScan < FOLDER_SCAN_INTERVAL and #cachedFolders > 0 then
        local valid = true
        for _, folder in ipairs(cachedFolders) do
            if not folder or not folder.Parent then
                valid = false
                break
            end
        end
        if valid then
            return cachedFolders
        end
    end
    
    lastFolderScan = now
    cachedFolders = {}
    
    for _, name in ipairs(MOB_FOLDER_NAMES) do
        for _, obj in ipairs(workspace:GetDescendants()) do
            if (obj:IsA("Folder") or obj:IsA("Model")) and obj.Name == name then
                table.insert(cachedFolders, obj)
            end
        end
    end
    
    return cachedFolders
end

local function getMobRootPart(mob)
    return mob:FindFirstChild("HumanoidRootPart") 
        or mob:FindFirstChild("Torso") 
        or mob:FindFirstChild("Head")
        or mob.PrimaryPart
end

local function getMobHeight(mob)
    local mobId = mob.Name
    if heightCache[mobId] then
        return heightCache[mobId]
    end
    
    local height = 5
    local humanoid = mob:FindFirstChildOfClass("Humanoid")
    if humanoid then
        height = humanoid.HipHeight * 2 + 1
    end
    
    heightCache[mobId] = height
    return height
end

local function getPlayerHRP()
    local char = player.Character
    if char then
        return char:FindFirstChild("HumanoidRootPart")
    end
    return nil
end

local function findClosestMob(hrp)
    local folders = findMobFolders()
    local hrpPos = hrp.Position
    
    local closest = nil
    local closestDist = CONFIG.radius
    
    for _, folder in ipairs(folders) do
        if folder and folder.Parent then
            for _, child in ipairs(folder:GetChildren()) do
                if child:IsA("Model") then
                    local humanoid = child:FindFirstChildOfClass("Humanoid")
                    local rootPart = getMobRootPart(child)
                    
                    if humanoid and rootPart and humanoid.Health > 0 then
                        local dist = (rootPart.Position - hrpPos).Magnitude
                        if dist <= closestDist then
                            closestDist = dist
                            closest = {
                                model = child,
                                humanoid = humanoid,
                                rootPart = rootPart,
                                height = getMobHeight(child)
                            }
                        end
                    end
                end
            end
        end
    end
    
    return closest
end

local function cleanup()
    if bodyPosition then
        bodyPosition:Destroy()
        bodyPosition = nil
    end
    if bodyGyro then
        bodyGyro:Destroy()
        bodyGyro = nil
    end
end

local function createBodyMovers(hrp)
    cleanup()
    
    local bp = Instance.new("BodyPosition")
    bp.MaxForce = Vector3.new(1e6, 1e6, 1e6)
    bp.P = 50000
    bp.D = 1000
    bp.Parent = hrp
    
    local bg = Instance.new("BodyGyro")
    bg.MaxTorque = Vector3.new(1e6, 1e6, 1e6)
    bg.P = 50000
    bg.D = 1000
    bg.Parent = hrp
    
    return bp, bg
end

local function isTargetValid(hrp)
    if not currentTarget then return false end
    if not currentTarget.model or not currentTarget.model.Parent then return false end
    if not currentTarget.humanoid then return false end
    
    local success, health = pcall(function()
        return currentTarget.humanoid.Health
    end)
    if not success or health <= 0 then return false end
    
    if not currentTarget.rootPart or not currentTarget.rootPart.Parent then return false end
    
    local dist = (currentTarget.rootPart.Position - hrp.Position).Magnitude
    if dist > CONFIG.radius * 1.5 then return false end
    
    return true
end

local function onFrame()
    if not CONFIG.enabled then
        cleanup()
        currentTarget = nil
        return
    end
    
    local hrp = getPlayerHRP()
    if not hrp then return end
    
    if not isTargetValid(hrp) then
        currentTarget = nil
    end
    
    local now = tick()
    
    if not currentTarget then
        if now - lastTargetSearch >= TARGET_SEARCH_INTERVAL then
            lastTargetSearch = now
            currentTarget = findClosestMob(hrp)
        end
    end
    
    if currentTarget then
        if not bodyPosition or bodyPosition.Parent ~= hrp then
            bodyPosition, bodyGyro = createBodyMovers(hrp)
        end
        
        local mobPos = currentTarget.rootPart.Position
        local mobHeight = currentTarget.height
        
        local totalHeight = (mobHeight / 2) + CONFIG.height
        local targetPos = mobPos + Vector3.new(0, totalHeight, 0)
        
        bodyPosition.Position = targetPos
        
        local lookAt = CFrame.lookAt(targetPos, mobPos)
        local rotatedDown = lookAt * CFrame.Angles(math.rad(90), 0, 0)
        bodyGyro.CFrame = rotatedDown
    else
        cleanup()
    end
end

player.CharacterAdded:Connect(function()
    cleanup()
    currentTarget = nil
    heightCache = {}
    cachedFolders = {}
    lastFolderScan = 0
end)

    Desc = "Auto Shoots Mobs With Guns",
    Default = false,
    Callback = function(state)
        gunAuraRunning = state
        if gunAuraRunning then
            _G.DelayShot = 0.4
            _G.ReachShot = 250
            _G.ModsAntilag = {}

            if gunAuraModsAntilagCon then
                gunAuraModsAntilagCon:Disconnect()
                gunAuraModsAntilagCon = nil
            end
            gunAuraModsAntilagCon = workspace.DescendantAdded:Connect(function(v)
                if v:IsA("Model") and v:FindFirstChild("HumanoidRootPart") and v:FindFirstChild("Humanoid") and v:FindFirstChild("Head") and not game.Players:GetPlayerFromCharacter(v) then
                    if v.Name == "Model_Unicorn" or v.Name == "Model_Horse" then return end
                    if v.Humanoid.Health > 0 then
                        table.insert(_G.ModsAntilag, v)
                    end
                end
            end)
            local TELEPORT_OFFSET = Vector3.new(0, 0, -2)

            gunAuraTask = task.spawn(function()
                while gunAuraRunning do
                    local DistanceGunAura, ModsTargetShotHead, ModsTargetShotHumanoid = math.huge, nil, nil
                    for i, v in pairs(_G.ModsAntilag) do
                        if v:IsA("Model") and v:FindFirstChild("HumanoidRootPart") and v:FindFirstChild("Humanoid") and v:FindFirstChild("Head") and not game.Players:GetPlayerFromCharacter(v) then
                            if v.Name == "Model_Unicorn" or v.Name == "Model_Horse" then
                                continue
                            end
                            local DistanceGun = (game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart").Position - v.HumanoidRootPart.Position).Magnitude
                            if DistanceGun < DistanceGunAura and DistanceGun < _G.ReachShot and v.Humanoid.Health > 0 then
                                ModsTargetShotHead, ModsTargetShotHumanoid, DistanceGunAura = v:FindFirstChild(_G.CharacterMods or "Head"), v.Humanoid, DistanceGun
                            end
                        end
                    end
                    if ModsTargetShotHead and ModsTargetShotHumanoid then
                        _G.ModsShotgun = {}
                        local ShotNow = {14, 8, 2, 5, 11, 17}
                        for i, v in pairs(game.Players.LocalPlayer.Character:GetChildren()) do
                            if v:FindFirstChild("ClientWeaponState") and v.ClientWeaponState:FindFirstChild("CurrentAmmo") then
                                if v.ClientWeaponState.CurrentAmmo.Value ~= 0 then
                                    if v.Name == "Shotgun" or v.Name == "Sawed-Off Shotgun" then
                                        for _, index in pairs(ShotNow) do
                                            _G.ModsShotgun[index] = ModsTargetShotHumanoid
                                        end
                                    else
                                        _G.ModsShotgun["2"] = ModsTargetShotHumanoid
                                    end
                                    local behindHead = ModsTargetShotHead.Position + TELEPORT_OFFSET
                                    game.ReplicatedStorage.Remotes.Weapon.Shoot:FireServer(
                                        game.Workspace:GetServerTimeNow(),
                                        v,
                                        CFrame.new(behindHead, ModsTargetShotHead.Position),
                                        _G.ModsShotgun
                                    )
                                    game.ReplicatedStorage.Remotes.Weapon.Reload:FireServer(game.Workspace:GetServerTimeNow(), v)
                                end
                            end
                        end
                    end
                    task.wait(_G.DelayShot)
                end
            end)
        else
            if gunAuraTask then
                task.cancel(gunAuraTask)
                gunAuraTask = nil
            end
            if gunAuraModsAntilagCon then
                gunAuraModsAntilagCon:Disconnect()
                gunAuraModsAntilagCon = nil
            end
        end
    end
})

local function getHorseNameList()
    local horses = {}
    local function findHorse(model, pathLabel)
        if model and model:IsA("Model") and model.Name == "Model_Horse" then
            local seat = model:FindFirstChildWhichIsA("VehicleSeat")
            if seat then
                local displayName = pathLabel or model:GetFullName()
                table.insert(horses, { Name = displayName, Model = model, Seat = seat })
            end
        end
    end
    findHorse(workspace:FindFirstChild("Model_Horse"), "Workspace Horse")
    local baseplates = workspace:FindFirstChild("Baseplates")
    if baseplates then
        for _, plateName in ipairs({"Baseplate16", "Baseplate24", "Baseplate30"}) do
            local baseplate = baseplates:FindFirstChild(plateName)
            if baseplate then
                local center = baseplate:FindFirstChild("CenterBaseplate")
                if center then
                    local animals = center:FindFirstChild("Animals")
                    if animals then
                        findHorse(animals:FindFirstChild("Model_Horse"), plateName)
                    end
                end
            end
        end
    end
    local nameList, modelMap = {}, {}
    for _, entry in pairs(horses) do
        table.insert(nameList, entry.Name)
        modelMap[entry.Name] = entry
    end
    table.insert(nameList, 1, "None")
    return nameList, modelMap
end

local function getRemote(name)
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    local r = ReplicatedStorage:FindFirstChild("Remotes")
    if r and r:FindFirstChild(name) then
        return r[name]
    end
    local shared = ReplicatedStorage:FindFirstChild("Shared")
    if shared then
        local universe = shared:FindFirstChild("Universe")
        if universe then
            local network = universe:FindFirstChild("Network")
            if network then
                local remotePromise = network:FindFirstChild("RemotePromise")
                if remotePromise then
                    local remotes = remotePromise:FindFirstChild("Remotes")
                    if remotes and remotes:FindFirstChild(name) then
                        return remotes[name]
                    end
                end
            end
        end
    end
    return nil
end

    local humanoid = character:FindFirstChild("Humanoid") or character:FindFirstChildOfClass("Humanoid")

    local nearestSeat, shortestDist = nil, math.huge
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("Seat") or obj:IsA("VehicleSeat") then
            local dist = (obj.Position - targetPosition).Magnitude
            if dist < shortestDist and dist <= 25 then
                nearestSeat, shortestDist = obj, dist
            end
        end
    end
    if nearestSeat then
        humanoid.Sit = false
        task.wait(0.2)
        root.CFrame = nearestSeat.CFrame + Vector3.new(0, 2, 0)
        task.wait(0.2)
        humanoid.Sit = true
    end
end

local function findByPath(root, relPath)
    local ptr = root
    for seg in string.gmatch(relPath, "[^%.]+") do
        ptr = ptr:FindFirstChild(seg)
        if not ptr then return nil end
    end
    return ptr
end

local function createOrUpdateESP(mobModel, color, label)
    if not mobModel or not mobModel:IsA("Model") then return end
    local targetPart = mobModel:FindFirstChild("Head") or mobModel.PrimaryPart or mobModel:FindFirstChildWhichIsA("BasePart")
    if not targetPart then return end
    local hl = mobModel:FindFirstChild("MobESPHL")
    if not hl then
        hl = Instance.new("Highlight")
        hl.Name = "MobESPHL"
        hl.Parent = mobModel
        hl.Adornee = mobModel
    end
    hl.FillColor = color
    hl.OutlineColor = Color3.fromRGB(255,255,255)
    hl.FillTransparency = 0.25
    hl.OutlineTransparency = 0
    local bbName = "ZinerHubESPLabel"
    local bb = targetPart:FindFirstChild(bbName)
    if not bb then
        bb = Instance.new("BillboardGui")
        bb.Name = bbName
        bb.Adornee = targetPart
        bb.AlwaysOnTop = true
        bb.Size = UDim2.new(0, 80, 0, 18)
        bb.StudsOffset = Vector3.new(0, 3.5, 0)
        bb.Parent = targetPart
        local txt = Instance.new("TextLabel")
        txt.Name = "Label"
        txt.AnchorPoint = Vector2.new(0.5, 0.5)
        txt.Position = UDim2.new(0.5, 0, 0.5, 0)
        txt.Size = UDim2.new(1, 0, 1, 0)
        txt.BackgroundTransparency = 1
        txt.TextStrokeTransparency = 0.3
        txt.Font = Enum.Font.SourceSansBold
        txt.TextScaled = true
        txt.Text = label or mobModel.Name
        txt.TextColor3 = color
        txt.Parent = bb
    else
        local txt = bb:FindFirstChild("Label")
        if txt then
            txt.Text = label or mobModel.Name
            txt.TextColor3 = color
        end
    end
end
